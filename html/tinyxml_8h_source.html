320  // Get a character, while interpreting entities.
321  // The length can be from 0 to 4 bytes.
322  inline static const char* GetChar(const char* p, char* _value, int* length, TiXmlEncoding encoding)
323  {
324  assert(p);
325  if (encoding == TIXML_ENCODING_UTF8) {
326  *length = utf8ByteTable[*((const unsigned char*)p)];
327  assert(*length >= 0 && *length < 5);
328  } else {
329  *length = 1;
330  }
331 
332  if (*length == 1) {
333  if (*p == '&')
334  return GetEntity(p, _value, length, encoding);
335  *_value = *p;
336  return p + 1;
337  } else if (*length) {
338  // strncpy( _value, p, *length ); // lots of compilers don't like this function (unsafe),
339  // and the null terminator isn't needed
340  for (int i = 0; p[i] && i < *length; ++i) {
341  _value[i] = p[i];
342  }
343  return p + (*length);
344  } else {
345  // Not valid text.
346  return 0;
347  }
348  }
